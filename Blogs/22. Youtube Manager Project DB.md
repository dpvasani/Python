## âœï¸ 3. Insert Data into the Table


## ğŸ” 8. Use `with` Statement (Auto Close & Commit)


> ğŸš€ Best practice: Ensures connection closes automatically.

# ğŸ’¼ Build a Mini CRUD App with SQLite + Python CLI

```python
import sqlite3

def connect():
    return sqlite3.connect("crud_app.db")

def create_table():
    with connect() as conn:
        conn.execute('''CREATE TABLE IF NOT EXISTS notes (
            id INTEGER PRIMARY KEY,
            title TEXT NOT NULL,
            content TEXT
        )''')

def add_note(title, content):
    with connect() as conn:
        conn.execute("INSERT INTO notes (title, content) VALUES (?, ?)", (title, content))

def view_notes():
    with connect() as conn:
        notes = conn.execute("SELECT * FROM notes").fetchall()
        for note in notes:
            print(note)

def update_note(note_id, title, content):
    with connect() as conn:
        conn.execute("UPDATE notes SET title=?, content=? WHERE id=?", (title, content, note_id))

def delete_note(note_id):
    with connect() as conn:
        conn.execute("DELETE FROM notes WHERE id=?", (note_id,))

# Usage
create_table()
add_note("First Note", "This is a test.")
view_notes()
update_note(1, "Updated Note", "Changed the content.")
delete_note(1)
```
